 Table users {
  id  int [pk, increment]
  email varchar

  username varchar
  full_name varchar

  password varchar
  avatar_url int

  activated boolean
  activation_link varchar
  blocked boolean

  theme varchar
  notifications boolean
  timezone varchar

  file_store_limit_bytes bigint
  file_store_used_bytes bigint

  updated_at timestamp
  created_at timestamp
}


Table medias {
  id int [pk, increment]
  user_id int
  name varchar
  thumbnail_url varchar
  content_url varchar
  created_at timestamp
}
Ref: users.id < medias.user_id


Table projects {
  id int [pk, increment]
  user_id int

  name varchar
  status varchar

  sort_by varchar

  updated_at timestamp
  created_at timestamp
}
Ref: projects.user_id > users.id


Table sections {
  id int [pk, increment]
  project_id int

  name varchar

  updated_at timestamp
  created_at timestamp

}
Ref: sections.project_id > projects.id


Table tasks {
  id int [pk, increment]
  project_id int
  section_id int

  name varchar
  description varchar

  priority int
  done boolean

  scheduled_at timestamp
  deadline_at timestamp

  updated_at timestamp
  created_at timestamp
}

Ref: tasks.project_id > projects.id
Ref: tasks.section_id > sections.id
Ref: tasks.id < medias.id
